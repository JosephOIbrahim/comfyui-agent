## Step 6: Write the ComfyUI bridge module

Create `agent/tools/comfyui_bridge.py` with this content:

```python
"""
ComfyUI Bridge — Connects comfyui-agent to a running ComfyUI instance.
Handles: API communication, node schema discovery, file transfer, model inventory.
"""

import json
import os
import urllib.request
import urllib.parse
from pathlib import Path


class ComfyUIBridge:
    def __init__(self, host=None, port=None, comfyui_path=None):
        self.host = host or os.environ.get("COMFYUI_HOST", "127.0.0.1")
        self.port = port or int(os.environ.get("COMFYUI_PORT", "8188"))
        self.comfyui_path = Path(comfyui_path or os.environ.get("COMFYUI_PATH", ""))
        self.base_url = f"http://{self.host}:{self.port}"
        self._node_schemas = None

    def is_connected(self):
        try:
            req = urllib.request.Request(f"{self.base_url}/system_stats")
            resp = urllib.request.urlopen(req, timeout=5)
            return resp.status == 200
        except Exception:
            return False

    def get_system_stats(self):
        return self._get("/system_stats")

    def get_node_schemas(self, force_refresh=False):
        if self._node_schemas and not force_refresh:
            return self._node_schemas
        self._node_schemas = self._get("/object_info")
        return self._node_schemas

    def get_node_schema(self, class_type):
        return self.get_node_schemas().get(class_type)

    def find_nodes_by_category(self, keyword):
        results = []
        for name, info in self.get_node_schemas().items():
            cat = info.get("category", "").lower()
            if keyword.lower() in cat or keyword.lower() in name.lower():
                results.append({"class_type": name, "category": info.get("category", "")})
        return results

    def validate_workflow(self, workflow):
        schemas = self.get_node_schemas()
        errors = []
        for node_id, node in workflow.items():
            ct = node.get("class_type", "UNKNOWN")
            if ct not in schemas:
                errors.append(f"Node {node_id}: unknown class_type '{ct}'")
            else:
                required = schemas[ct].get("input", {}).get("required", {})
                for inp in required:
                    if inp not in node.get("inputs", {}):
                        errors.append(f"Node {node_id} ({ct}): missing input '{inp}'")
        return errors

    def list_models(self, model_type="checkpoints"):
        schemas = self.get_node_schemas()
        loaders = {
            "checkpoints": "CheckpointLoaderSimple",
            "loras": "LoraLoader",
            "vae": "VAELoader",
            "controlnet": "ControlNetLoader",
        }
        loader = loaders.get(model_type)
        if not loader or loader not in schemas:
            return []
        required = schemas[loader].get("input", {}).get("required", {})
        for _, spec in required.items():
            if isinstance(spec, list) and spec and isinstance(spec[0], list):
                return spec[0]
        return []

    def queue_workflow(self, workflow, client_id="comfyui-agent"):
        return self._post("/prompt", {"prompt": workflow, "client_id": client_id})

    def get_queue(self):
        return self._get("/queue")

    def get_history(self, prompt_id=None):
        path = f"/history/{prompt_id}" if prompt_id else "/history"
        return self._get(path)

    def upload_image(self, filepath, subfolder="agent_inputs"):
        import mimetypes
        boundary = "----AgentBoundary"
        filename = os.path.basename(filepath)
        mime = mimetypes.guess_type(filepath)[0] or "image/png"
        with open(filepath, "rb") as f:
            file_data = f.read()
        body = (
            f"--{boundary}\r\nContent-Disposition: form-data; name=\"image\"; "
            f"filename=\"{filename}\"\r\nContent-Type: {mime}\r\n\r\n"
        ).encode() + file_data + (
            f"\r\n--{boundary}\r\nContent-Disposition: form-data; name=\"subfolder\""
            f"\r\n\r\n{subfolder}\r\n--{boundary}\r\nContent-Disposition: form-data; "
            f"name=\"type\"\r\n\r\ninput\r\n--{boundary}--\r\n"
        ).encode()
        req = urllib.request.Request(
            f"{self.base_url}/upload/image", data=body,
            headers={"Content-Type": f"multipart/form-data; boundary={boundary}"},
        )
        return json.loads(urllib.request.urlopen(req).read())

    def download_output(self, filename, subfolder="", save_to=None):
        params = urllib.parse.urlencode({"filename": filename, "subfolder": subfolder, "type": "output"})
        data = urllib.request.urlopen(f"{self.base_url}/view?{params}").read()
        if save_to:
            os.makedirs(os.path.dirname(save_to), exist_ok=True)
            with open(save_to, "wb") as f:
                f.write(data)
        return data

    def free_vram(self):
        return self._post("/free", {"unload_models": True, "free_memory": True})

    def _get(self, path):
        return json.loads(urllib.request.urlopen(f"{self.base_url}{path}", timeout=30).read())

    def _post(self, path, data):
        payload = json.dumps(data).encode()
        req = urllib.request.Request(
            f"{self.base_url}{path}", data=payload,
            headers={"Content-Type": "application/json"},
        )
        return json.loads(urllib.request.urlopen(req).read())


if __name__ == "__main__":
    bridge = ComfyUIBridge()
    if bridge.is_connected():
        print("✓ ComfyUI is running")
        stats = bridge.get_system_stats()
        devices = stats.get("devices", [{}])
        if devices:
            d = devices[0]
            vram = d.get("vram_total", 0) / (1024**3)
            print(f"  GPU: {d.get('name', '?')} ({vram:.0f}GB VRAM)")
        for mt in ["checkpoints", "loras", "vae"]:
            print(f"  {mt}: {len(bridge.list_models(mt))} installed")
        schemas = bridge.get_node_schemas()
        print(f"  Total node types: {len(schemas)}")
    else:
        print("✗ ComfyUI not running at", bridge.base_url)
```

```markdown
# /project:analyze-video — ANALYST Expert

You are the ANALYST. Your job: take a reference video → produce a structured storyboard.

Input: $ARGUMENTS — a video URL or local file path.

Do these steps in order. Run every command yourself.

1. Download or locate the video:
   - If it's a URL: `curl -L -o workspace/reference/input_video.mp4 "$URL"`
   - If it's a local path: `cp "$PATH" workspace/reference/input_video.mp4`

2. Get metadata:
   `ffprobe -v quiet -print_format json -show_format -show_streams workspace/reference/input_video.mp4`

   `ffmpeg -i workspace/reference/input_video.mp4 -vf "select='gt(scene,0.3)',showinfo" -vsync vfr workspace/keyframes/scene_%04d.png 2>&1 | grep showinfo`
   Parse timestamps from the output. If < 3 scenes, lower threshold to 0.2. If > 12, raise to 0.4.

4. Extract first + last frame of each segment:
   `ffmpeg -ss {start} -i workspace/reference/input_video.mp4 -frames:v 1 workspace/keyframes/seg_{N}_first.png`
   `ffmpeg -ss {end - 0.04} -i workspace/reference/input_video.mp4 -frames:v 1 workspace/keyframes/seg_{N}_last.png`

5. Look at each keyframe and describe: shot type, camera, style, subject, mood, lighting.

6. Save structured storyboard to `workspace/storyboard.json` with fields:
   source (path, duration, fps, resolution), segments array (id, start, end, duration_seconds, shot_description, camera_type, style_notes, first_frame_path, last_frame_path, prompt_draft).

7. Show a summary table and ask: "Storyboard ready. Proceed to building workflows?"
```

```markdown
# /project:assemble-montage — EDITOR Expert

You are the EDITOR. Your job: take generated clips → assemble into final montage.

Prerequisites: clips in workspace/outputs/. Storyboard for ordering.

1. Probe each clip for resolution/fps:
   `ffprobe -v quiet -print_format json -show_streams workspace/outputs/seg_01.mp4`

   `ffmpeg -i workspace/outputs/seg_{N}.mp4 -vf "scale=1080:1920:force_original_aspect_ratio=decrease,pad=1080:1920:-1:-1,fps=24" -c:v libx264 -crf 18 -an workspace/outputs/norm_seg_{N}.mp4`

   Write workspace/outputs/concat_list.txt with lines like: `file 'norm_seg_01.mp4'`

4. Concatenate:
   `cd workspace/outputs && ffmpeg -f concat -safe 0 -i concat_list.txt -c:v libx264 -preset medium -crf 18 -pix_fmt yuv420p ../montage_v1.mp4`

5. Mix reference audio if available:
   `ffmpeg -i workspace/reference/input_video.mp4 -vn -acodec aac -b:a 192k workspace/reference/ref_audio.aac`
   `ffmpeg -i workspace/montage_v1.mp4 -i workspace/reference/ref_audio.aac -c:v copy -c:a aac -map 0:v -map 1:a -shortest workspace/montage_v1_audio.mp4`

6. Report duration and ask: "Montage assembled. Run QA comparison?"
```

```python
import os, sys, shutil, subprocess
from pathlib import Path

env_path = Path("config/bridge.env")
if env_path.exists():
    for line in env_path.read_text().splitlines():
        line = line.strip()
        if line and not line.startswith("#") and "=" in line:

def check(name, ok, detail=""):
    s = "✓" if ok else "✗"
    print(f"  {s} {name}" + (f" — {detail}" if detail else ""))
    return ok
```
